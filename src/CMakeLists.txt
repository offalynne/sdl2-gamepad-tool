set(SOURCES
    main.cpp
    ui/mainwindow.cpp
    ui/mainwindow.h
    core/sdlpollevent.cpp
    core/sdlpollevent.h
    core/GamepadMapper.cpp
    core/GamepadMapper.h
    core/GamepadDatabase.cpp
    core/GamepadDatabase.h
    core/Logger.cpp
    core/Logger.h
    core/JoystickEnumerator.cpp
    core/JoystickEnumerator.h
    network/filedownloader.cpp
    network/filedownloader.h
    platform/Environment.h
)

# Platform-specific environment source
if(APPLE)
    list(APPEND SOURCES platform/EnvironmentMac.cpp)
elseif(WIN32)
    list(APPEND SOURCES platform/EnvironmentWin.cpp)
else()
    list(APPEND SOURCES platform/EnvironmentLinux.cpp)
endif()

# Resources
set(RESOURCES resources/res.qrc)

# UI files
set(UI_FILES ui/mainwindow.ui)

add_executable(gamepad-tool WIN32 ${SOURCES} ${RESOURCES} ${UI_FILES})

# Include directories so headers can find each other
target_include_directories(gamepad-tool PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/ui
    ${CMAKE_CURRENT_SOURCE_DIR}/core
    ${CMAKE_CURRENT_SOURCE_DIR}/network
    ${CMAKE_CURRENT_SOURCE_DIR}/platform
    ${CMAKE_CURRENT_BINARY_DIR}  # for ui_mainwindow.h generated by AUTOUIC
)

# Windows resource file
if(WIN32)
    target_sources(gamepad-tool PRIVATE platform/gamepadtool.rc)
endif()

# Qt libraries
if(QT_VERSION_MAJOR EQUAL 6)
    target_link_libraries(gamepad-tool PRIVATE
        Qt6::Core Qt6::Gui Qt6::Widgets Qt6::Network
    )
else()
    target_link_libraries(gamepad-tool PRIVATE
        Qt5::Core Qt5::Gui Qt5::Widgets Qt5::Network
    )
endif()

# SDL2
if(TARGET SDL2::SDL2)
    target_link_libraries(gamepad-tool PRIVATE SDL2::SDL2)
elseif(TARGET SDL2::SDL2-static)
    target_link_libraries(gamepad-tool PRIVATE SDL2::SDL2-static)
elseif(TARGET PkgConfig::SDL2)
    target_link_libraries(gamepad-tool PRIVATE PkgConfig::SDL2)
endif()

# Platform defines
if(UNIX AND NOT APPLE)
    target_compile_definitions(gamepad-tool PRIVATE LINUX_BUILD)
endif()

# macOS bundle settings
if(APPLE)
    set_target_properties(gamepad-tool PROPERTIES
        MACOSX_BUNDLE TRUE
        MACOSX_BUNDLE_ICON_FILE gamepadtool
        MACOSX_BUNDLE_GUI_IDENTIFIER "local.generalarcade.gamepadtool"
        MACOSX_BUNDLE_BUNDLE_NAME "SDL2 Gamepad Tool"
        MACOSX_BUNDLE_BUNDLE_VERSION "${PROJECT_VERSION}"
        MACOSX_BUNDLE_SHORT_VERSION_STRING "${PROJECT_VERSION}"
        MACOSX_BUNDLE_INFO_PLIST "${CMAKE_CURRENT_SOURCE_DIR}/platform/Info.plist.in"
        XCODE_ATTRIBUTE_CODE_SIGN_ENTITLEMENTS "${CMAKE_CURRENT_SOURCE_DIR}/platform/gamepad-tool.entitlements"
    )
    # Bundle the app icon into Contents/Resources
    set(APP_ICON "${CMAKE_CURRENT_SOURCE_DIR}/resources/images/gamepadtool.icns")
    target_sources(gamepad-tool PRIVATE ${APP_ICON})
    set_source_files_properties(${APP_ICON} PROPERTIES
        MACOSX_PACKAGE_LOCATION Resources
    )
    # Bundle files alongside the executable in Contents/MacOS
    set(MACOS_BUNDLE_FILES
        "${CMAKE_CURRENT_SOURCE_DIR}/platform/local.generalarcade.gamepadtool.plist"
        "${CMAKE_CURRENT_SOURCE_DIR}/resources/gamecontrollerdb.txt"
    )
    target_sources(gamepad-tool PRIVATE ${MACOS_BUNDLE_FILES})
    set_source_files_properties(${MACOS_BUNDLE_FILES} PROPERTIES
        MACOSX_PACKAGE_LOCATION MacOS
    )
    # Run macdeployqt to bundle Qt and SDL2 into the app
    find_program(MACDEPLOYQT macdeployqt HINTS "${Qt6_DIR}/../../../bin" "${_qt_bin_dir}")
    if(MACDEPLOYQT)
        # Use CODESIGN_IDENTITY env var if set, otherwise ad-hoc sign
        set(CODESIGN_ID "$ENV{CODESIGN_IDENTITY}")
        if(NOT CODESIGN_ID)
            set(CODESIGN_ID "-")
        endif()
        set(ENTITLEMENTS "${CMAKE_CURRENT_SOURCE_DIR}/platform/gamepad-tool.entitlements")
        add_custom_command(TARGET gamepad-tool POST_BUILD
            COMMAND "${MACDEPLOYQT}" "$<TARGET_BUNDLE_DIR:gamepad-tool>"
            COMMAND codesign --force --deep --options runtime
                --entitlements "${ENTITLEMENTS}"
                --sign "${CODESIGN_ID}"
                "$<TARGET_BUNDLE_DIR:gamepad-tool>"
            VERBATIM
            COMMENT "Running macdeployqt to bundle Qt frameworks and signing"
        )
    else()
        message(WARNING "macdeployqt not found — Qt frameworks will not be bundled into the app")
    endif()
endif()

# Copy gamecontrollerdb.txt next to binary (Linux/Windows)
if(NOT APPLE)
    add_custom_command(TARGET gamepad-tool POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${CMAKE_CURRENT_SOURCE_DIR}/resources/gamecontrollerdb.txt"
            "$<TARGET_FILE_DIR:gamepad-tool>/gamecontrollerdb.txt"
    )
endif()

# Install rules (Linux)
if(UNIX AND NOT APPLE)
    include(GNUInstallDirs)
    install(TARGETS gamepad-tool DESTINATION ${CMAKE_INSTALL_BINDIR})
    install(FILES platform/gamepad-tool.desktop
        DESTINATION ${CMAKE_INSTALL_DATADIR}/applications)
    install(FILES resources/images/gamepad-tool.png
        DESTINATION ${CMAKE_INSTALL_DATADIR}/icons/hicolor/256x256/apps)
    install(FILES resources/gamecontrollerdb.txt
        DESTINATION ${CMAKE_INSTALL_DATADIR}/gamepad-tool)
endif()
